Q2

Null Hypothesis: Mean sleep total in sampled population is 7hrs

Alternative Hypothesis: Mean sleep total in sampled population is not equal to 7hrs

CI: 5%

observed_sleep_stat <- msleep %>% 
  summarise(
    mean_sleep = mean(sleep_total)
  )

null_distribution_msleep <- msleep %>% 
  specify(response = sleep_total) %>% 
  hypothesise(null = "point", mu = 7) %>% 
  generate(reps = 10000, type = "bootstrap") %>% 
  calculate(stat = "mean")


(p_value_sleep <- null_distribution_msleep %>% 
  get_pvalue(obs_stat = observed_sleep_stat,

             direction = "both"))

The p-value is less than 5% therefore we can reject the null hypothesis as the mean sleep in the sampled population does not suggest most animals sleep for 7 hrs a night

Q3

Null Hypothesis: Mean sleep total is the same for omnivores and herbivores

Alternative Hypothesis: Mean sleep total in sampled population is significantly different between omnivores and herbivores

CI: 5%

msleep %>% 
  distinct(vore)
msleep_vore <- msleep %>% 
  filter(vore %in% c("omni", "herbi")) 
obs_vore_stat <- msleep_vore %>% 
  specify(sleep_total ~ vore) %>% 
  calculate(stat = "diff in means",
            order = c("omni",
                      "herbi"))
null_dist_vore <- msleep_vore %>% 
  specify(sleep_total ~ vore) %>% 
  hypothesise(null = "independence") %>% 
  generate(reps = 10000, 
           type = "permute") %>% 
  calculate(stat = "diff in means",
            order = c("omni",
                      "herbi"))
null_dist_vore %>% 
  visualise(bins = 30) +
  shade_pvalue(obs_stat = obs_vore_stat,
               direction = "greater")

With a p-value of 0.12 we cant reject the null hypothesis, and so we can say the mean sleep for omnivores and herbivores is not significantly different

Q4

Null hypothesis: The proportion of domesticated animals in the population is 5%

Alternative hypothesis: The proportion of domesticated animals in the population is greater than 5%

CI: 5%

msleep_domestic <- msleep %>% 
  mutate(domesticated = case_when(
    conservation == "domesticated" ~ "Yes",
    TRUE ~ "No"
  ))
domestic_prop <- msleep_domestic %>% 
  summarise(prop =
    (sum(domesticated == "Yes"))/nrow(msleep_domestic)
  )
null_dom_distribution <- msleep_domestic %>% 
  specify(response = domesticated,
          success = "Yes") %>% 
  hypothesise(null = "point",
              p = 0.05) %>% 
  generate(reps = 30000,
           type = "draw") %>% 
  calculate(stat = "prop")
null_dom_distribution %>% 
  visualise() +
  shade_pvalue(obs_stat = domestic_prop,
               direction = "greater")
null_dom_distribution %>% 
  get_p_value(obs_stat = domestic_prop,
              direction = "greater")

With a p-value of 0.008 and a CI of 5% we can reject the null hypothesis that 5% of the animals are domesticated, and can say that significantly more than 5% of animals in the sample are domesticated.
